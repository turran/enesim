00099 static inline uint32_t enesim_renderer_gradient_reflect_color_get(Enesim_Color *src, int len, Eina_F16p16 p)
00100 {
00101     int fp;
00102     int fp_next;
00103     uint32_t v;
00104     uint16_t a;
00105 
00106     fp = eina_f16p16_int_to(p);
00107     fp = fp % (2 * len);
00108     if (fp < 0) fp += 2 * len;
00109     if (fp >= len) fp = (2 * len) - fp - 1;
00110     fp_next = (fp < (len - 1) ? (fp + 1) : len - 1);
00111 
00112     a = eina_f16p16_fracc_get(p) >> 8;
00113     v = argb8888_interp_256(1 + a, src[fp_next], src[fp]);
